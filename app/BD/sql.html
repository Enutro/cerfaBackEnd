CREATE TABLE apprenti ( apprenti_id INT NOT NULL AUTO_INCREMENT, nom_naissance VARCHAR(255) NOT NULL, nom_usage VARCHAR(255)NOT NULL, prenom VARCHAR(255)NOT NULL, nom_legal VARCHAR(255) NOT NULL, prenom_legal VARCHAR(255) NOT NULL, nationalite INT(255) NOT NULL, date_naissance DATE NOT NULL, sexe CHAR(1) NOT NULL, commune_naissance VARCHAR(255) NOT NULL, inscription_sportif_hn BOOLEAN NOT NULL, handicape BOOLEAN NOT NULL, situation INT(255) NOT NULL, diplome VARCHAR(255) NOT NULL, classe VARCHAR(255) NOT NULL, intitule_diplome VARCHAR(255) NOT NULL, diplome_eleve VARCHAR(255) NOT NULL, crea_entreprise BOOLEAN NOT NULL, numero INT(255) NOT NULL, rue VARCHAR(255) NOT NULL, complement VARCHAR(255), commune VARCHAR(255) NOT NULL, telephone VARCHAR(255) NOT NULL, courriel VARCHAR(255) NOT NULL, rue_r VARCHAR(255), complement_r VARCHAR(255), commune_r VARCHAR(255), telephone_r VARCHAR(255), courriel_r VARCHAR(255), PRIMARY KEY (apprenti_id) );
